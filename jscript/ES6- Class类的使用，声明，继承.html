    声明一个类

    //class 类
class Coder{
    // 类中都是方法 函数
    //val是name方法的参数
    name(val){
        console.log(val) //类
        return val;
    }
    //不要逗号
    skill(val){
        //this指向Coder
        console.log(this.name('调用name') + 'skill' + val); //调用nameskill技能 
    }
    //类的传参
    constructor(a,b){
        this.a = a;
        this.b = b;

    }
    add(){
        return this.a + this.b;
    }
}
    调用方法
    // 调用方法
let ananClass = new Coder;
ananClass.name('类');
ananClass.skill('技能');

     类传参

    // 类传参
let ananiah = new Coder(1,2);
console.log(ananiah.add()) //3

     

    继承

    // 类的继承
class anan extends Coder{

}
let an = new anan;
an.name('anan的继承')  //继承Coder中的name方法
