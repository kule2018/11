    1.for in 循环 并且使用hasOwnProperty 方法

     var jsonObject1 = {
        "name": "xiaoming",
        "age": 29
    },
    var keys1 = [];
    for (var p1 in jsonObject1) {
        if (jsonObject1.hasOwnProperty(p1))
            keys1.push(p1);
    }
    alert(keys1);

    2.如果不使用 hasOwnProperty，给Object 添加自己的属性时，也会输出出来
