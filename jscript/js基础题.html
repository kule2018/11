    　　　　// 第一步 正阶梯模式
    　　　　// 即  第一行 1  第二行 2  第三行 3 ......
    　　　　// 1.声明变量str  用来存储字符串
    　　　　let str = '';
    　　　　// 2. 利用循环将乘法表写出来
    　　　　for (let i = 1; i<=9;i++) {　　　　// 1 - 9  所以循环9次
    　　　　　　for(let j = 1;j<=i){　　　　　//  阶梯模式,所以令j<=i
    　　　　　　　let num = i * j;
    　　　　　　　let str += i +'×'+j+'='+num
    　　　　　　　document.write(str)
    　　　　　　}
    　　　　document.write(\n)　　　　　　
    　　　　}
