    <script>

    //js获取页面所有搜索条件
    function GetSearchData(SearchDivID) {

        var ObjID= SearchDivID? SearchDivID: ".search"; //默认为类样式.search， <div  class="search">
        var data = {};
        var $SearchObj = $(ObjID);

        $SearchObj.children().each(function () {
            var name = $(this).attr('name');
            if (name) {
                var val = $(this).val();
                if (val) {
                    data[name] = val;
                }
            }
        });

        return data;
    }

     

     

     

     

        function getlist() {

            var data = GetSearchData();
            $.post("/Controller/View", data,
                function (data) {
                    $("#ID").html(data);
                })

        }

     

      </script>
