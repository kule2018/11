    1、js 获取当前日期星期几

    getWeek(time) {
    let weekDay = ["周日","周一", "周二", "周三", "周四", "周五", "周六"];    
    let myDate = new Date(Date.parse(time));  
    return weekDay[myDate.getDay()]
},

console.log(getWeek('2019-09-11')   // 周三
