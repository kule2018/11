    一、es6 Set去重

    function removal(arr) {
  return Array.from(new Set(arr))
}

       let arr=[1,2,1,3,4,5,5]

       removal(arr)//[1, 2, 3, 4]


    二、利用 filter
       function removal(arr){
            return arr.filter((item,index,arr)=>{
              return arr.indexOf(item,0) == index;

           })
      }

       let arr=[1,2,1,3,4,5,5]

       removal(arr)//[1, 2, 3, 4]

    三、利用 reduce 

    let newArr = arr.reduce((prev, cur) => {
    prev.indexOf(cur) === -1 && prev.push(cur);
    return prev;
},[]);
