    3.单例模式

    就只有一个对象存在

    实例：

    const single =(function(){
    let status; //用它来存储状态
    let num =0;
    function init(){
    return{
    sayHello:function(){
    console.log("hello,world");
    },
    sayGoodBye:function(){
    console.log("再见");
    },
    setNum:function(){
    return num++;
    },
    getNum:function(){
    return num;
    }
    }
    }
    return{
    getStatus:function(){
    if(!status){
    status =init();
    }
    return status;
    }
    }
    })();

    // console.log(single);
    // console.log(single.getStatus());
    let info=single.getStatus();

    console.log(info.setNum());
    console.log(info.setNum());
    console.log(info.setNum());
    console.log(info.setNum());

    let info2=single.getStatus();
    console.log("------------"+info2.getNum());
    info2.setNum();
