     1 // Data Set
 2 // One top level comment 
 3 var comments = [{
 4     id: 1,
 5     parent_id: null

20 const nest = (items, id = null, link = 'parent_id') =>
21   items
22     .filter(item => item[link] === id)
23     .map(item => ({ ...item, children: nest(items, item.id) }));
