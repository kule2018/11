                        var  v  =  Math.random ( );     

                         Math.ceil (v*10 );           -- 取1——10的随机数

     a. match ( b )     -- 判断是否匹配，

            var a = "aaa"
            var b = "aaa"
            if (aa.match(aaa))

             {                        -- 两字符串进行匹配，如果不匹配返回 null，如果相当返回子符                                                                                                          
                alert("相等")
              }
            else

             {                              -- 只是判断内容是否匹配，不看数据类型。

    事件冒泡：

         当元素嵌套的时候，内部元素激发某个事件后，默认情况下外部元素相应的事件也会跟着触发

      阻止冒泡：

                 window.event ? window.event.cancelBubble = true : e.stopPropagation();
