for (初始化表达式1; 判断表达式2; 自增表达式3) {

}

    2.for...in索引遍历

     

    var obj1 = {
    name:'张三',
    age : 17,
    sex : '男',
}

for(var k in obj1){
    console.log(k);
    console.log(obj1[k]);
}

     

    注意：使用for …in语法，同样可以遍历数组

    注意：如果属性名或方法名，是一个变量，则使用对象[变量名] 语法

    3.for...of值遍历

     

    //遍历数组
var team = ["师父", "大师兄", "二师兄", "沙师弟", "小白龙"];
for(var v of team){
      console.log(v);
}
//也可以遍历字符串
var str = "zhangsan";
for(var v of str){
      console.log(v);
}

     

    注意：不能遍历对象
