    //验证手机号
    function check_lxdh(lxdh){
        var mobile  = /^(13[0-9]|14[579]|15[0-3,5-9]|16[6]|17[0135678]|18[0-9]|19[89])\d{8}$/;  
        if(mobile.test(lxdh)){ 
                return true;
        }else{
                return false;
        }
}

    //验证手机号
function check_sfz    (sfzhm){
        var  identity_card  =/^[1-9]\d{5}(18|19|20)?\d{2}(((0[13578]|1[02])(0[1-9]|[12][0-9]|3[01]))|((0[469]|11)(0[1-9]|[12][0-9]|30))|(02(0[1-9]|[1][0-9]|2[0-8])))\d{3}(\d|[xX])?$/;
        if(identity_card.test(sfzhm)){ 
                return true;
        }else{
                return false;
        }
}
    //验证是否数字　
function check_number(val){
　 var regPos = /^\d+(\.\d+)?$/; //非负浮点数 
   var regNeg = /^(-(([0-9]+\.[0-9]*[1-9][0-9]*)|([0-9]*[1-9][0-9]*\.[0-9]+)|([0-9]*[1-9][0-9]*)))$/; //负浮点数
　　if(regPos.test(val) || regNeg.test(val)) { 
        return true;
    } else { 
        return false;
    } 
}
