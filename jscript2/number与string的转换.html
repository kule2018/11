/*
var num = 10;
var res = num + ""
alert(num + ", " + typeof (num));
alert(res + ", " + typeof (res));

// 使用String(数字)函数
/*
var num = 10;
var res = String(num);
alert(num + ", " + typeof (num));

*/

// 没有固定精度的表示
/*
var n = 1234.56789;
var s4 = n.toFixed(2);

var s6 = n.toPrecision(2); // 有效位数

alert(s4 + ", " + typeof(s4));

alert(s6 + ", " + typeof(s6));
*/
