    • 点击除指定区域外的空白处，隐藏div
    原文出处:本文由博客园博主小锐是梦想家提供。
    原文连接:https://www.cnblogs.com/haorui3/p/11067942.html

    <script>
$(document).click(function (e) {
    var $target = $(e.target);
    //点击.zanpl和.quanzipl以外的地方 隐藏.quanzipl和footer
    if(!$target.is('.zanpl *') && !$target.is('.quanzipl *')) { 
       $(    '.quanzipl').hide(); 
       $(    'footer').show(); }  
    }); 

    </script>
